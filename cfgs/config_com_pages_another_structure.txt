server{
    listen localhost:2002;
    server_name example.com;
	allow_methods GET POST DELETE
    root web/pages/;

	location / {
		index index.html
		auto_index off
	}

	location /pages {
		root web/pages/
	}

	location /images {
		root web/images/
		auto_index on
	}

	location /upload {
		root web/cgi-bin/;
      	cgi_path /bin/bash;
      	cgi_ext .sh;
		upload_store web/images/
	}

	location /about {
		index index.html
	}

	location /cgi-bin {
      root web/cgi-bin/;
      cgi_path /bin/bash;
      cgi_ext .sh;
  	}
}
