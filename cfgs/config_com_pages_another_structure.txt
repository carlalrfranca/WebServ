server{
    listen localhost:2007;
	allow_methods GET POST DELETE
    root web/pages/;
	client_max_body_size 20M

	location / {
		index index.html
		auto_index off
	}

	location /images {
		root web/images/
		auto_index on
	}

	location /upload {
		root web/cgi-bin/;
      	cgi_path /bin/bash;
      	cgi_ext .sh;
		upload_store web/images/
	}

	location /about {
		index index.html
		return 307 tourII.html
	}

	location /cgi-bin {
      root web/cgi-bin/;
      cgi_path /bin/bash;
      cgi_ext .sh;
  	}

}
