server{
    listen localhost:2007;
    server_name example.com;
	allow_methods GET POST DELETE
    root web/pages/;

	location / {
		index index.html
		auto_index off
	}

	location /images {
		root web/images/
		auto_index on
		allow_methods GET
	}

	location /upload {
		root web/cgi-bin/;
      	cgi_path /bin/bash;
      	cgi_ext .sh;
		upload_store web/images/
	}

	location /about {
		index index.html
	}

	location /cgi-bin {
      root web/cgi-bin/;
      cgi_path /bin/bash;
      cgi_ext .sh;
	  
  	}
}
