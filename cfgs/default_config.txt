server{
    listen 127.0.0.1:2002;
	server_name	leticia;
	allow_methods GET POST DELETE
    root web/pages/;
	client_max_body_size 10M

	location / {
		index index.html
		auto_index off
		cgi_path /usr/bin/python3;
      	cgi_ext .py;
	}

	location /images {
		root web/images/
		auto_index on
	}

	location /upload {
		root web/cgi-bin/;
      	cgi_path /usr/bin/python3;
      	cgi_ext .py;
		upload_store web/images/
	}

	location /about {
		index index.html
		return 307 tourII.html
	}

	location /cgi-bin {
      root web/cgi-bin/;
      cgi_path /usr/bin/python3;
      cgi_ext .py;
	  upload_store web/images/
  	}
}
